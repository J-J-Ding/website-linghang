/**
 * @brief 将GHao协议端口号转换为系统内部端口号
 *
 * @param usPort GHao协议格式的4字节端口号
 * @return UINT16 转换后的系统内部端口号，无效返回PORT_INVALID
 */
UINT16 sspNMGHaoPort2SwitchPort(UINT32 usPort)//用于ghao解析接收自L1的4字节端口号
{
    UINT8  ucHighValue = 0;          /*端口号的高8位*/
    UINT8  ucLowValue = 0;           /*端口号的低8位*/
    UINT16 usL2Port = 0;            /* 二层端口号 */

    ucHighValue  = (usPort & 0xFFFF0000) >> 16;
    ucLowValue   = usPort & 0x0000FFFF;

    if(ucHighValue == LINE_PORT_HIGH_8)
    {
        usL2Port = SSP_DEVICE_LINE_MIN + ucLowValue - 1;
        if((usL2Port > SSP_DEVICE_OSU_LINE_MAX) || (usL2Port < SSP_DEVICE_LINE_MIN))
        {
            usL2Port = PORT_INVALID;
            return usL2Port;
        }
        return usL2Port;
    }

    return PORT_INVALID;
}

/**
 * @brief 计算线路端口速率参数
 *
 * @param usCmdAct 操作类型：F508_IP_CMDACT_DEL/F508_IP_CMDACT_ADD
 * @param ptLinePortCfg 端口配置结构指针
 * @param usPortId 输出端口号
 * @param ulCir 输出承诺信息速率（kbps）
 * @param usAct 输出操作动作类型
 * @return INT32 执行结果：SSP_OK成功，其他值失败
 */
INT32 sspLinePortSpeedCalc(UINT16 usCmdAct, SSP_PORT_LINE_CFG_S *ptLinePortCfg, UINT16 *usPortId, UINT32 *ulCir, UINT16 *usAct)
{
    switch (usCmdAct)
    {
        case F508_IP_CMDACT_DEL:
        {
            g_IPortgOsuId[ptLinePortCfg->m_usNMPort] = 0;
            *ulCir = LINE_PORT_DEFAULT_N * IP_RATE_CONVERT;  /*删除时将限速值恢复为带宽最大值10G*/
            *usAct = PORT_LINE_ACT_DEL;
            break;
        }
        case F508_IP_CMDACT_ADD:
        {
            g_IPortgOsuId[ptLinePortCfg->m_usNMPort] = ptLinePortCfg->m_usOsuId;
            *ulCir = ptLinePortCfg->m_usNValue * IP_RATE_CONVERT; /*10400 目前N制转换为N 1=10.4M*/
            *usAct = PORT_LINE_ACT_ADD;
            break;
        }
        default:
        {
            PORT_LOG_ERR("cmd_type %d error!", usCmdAct);
            return SSP_ERROR;
        }
    }

    *usPortId = ptLinePortCfg->m_usNMPort + (SSP_DEVICE_LINE_MIN - 1);
    PORT_LOG_NOTICE("Calc F508 Act %d, Cir %d, Port %d!", *usAct, *ulCir, *usPortId);
    return SSP_OK;
}

/**
 * @brief 解析端口名称字符串中的端口号
 *
 * @param ifName 接口名称字符串（如"VCG-0/1/2"）
 * @return UINT16 解析出的端口号，无效返回0
 */
static UINT16 ParsePortNumber(const CHAR *ifName)
{
    const CHAR *pDash = NULL;
    const CHAR *pSlash = NULL;

    if (ifName == NULL)
    {
        return 0;
    }

    /* 查找最后一个'-'的位置 */
    pDash = strrchr(ifName, '-');
    if (pDash == NULL)
    {
        return 0;
    }

    /* 在'-'之后查找最后一个'/' */
    pSlash = strrchr(pDash, '/');
    if (pSlash == NULL)
    {
        return 0;
    }

    /* 转换最后部分为数字 */
    char *endPtr = NULL;
    errno = 0;
    long num = strtol(pSlash + 1, &endPtr, 10);

    /* 错误检查 */
    if (errno != 0 || endPtr == pSlash + 1 || *endPtr != '\0')
    {
        return 0;  // 无效格式返回0
    }

    /* 范围检查 */
    if (num < 0 || num > SSP_DEVICE_LINE_NUM)
    {
        return 0;
    }

    return (UINT16)num;
}

    UINT32 obj_index = 0;
    UINT16 obj_act = 0;

                tUobject_attr.m_bandwidth += tUobject_attr.m_bandwidth/VC12_SPEED_INCREASE;
            }
            else
            {
                tUobject_attr.m_bandwidth += tUobject_attr.m_bandwidth/VC4_SPEED_INCREASE;
            }
            PORT_LOG_ERR("VCG bandwidth = %d", tUobject_attr.m_bandwidth);
        }

        /* 转换动作类型并设置端口参数 */
        converted_act = ConvertActionType(obj_act);
        iRet = sspLinePortSet(usPort + SSP_DEVICE_LINE_MIN - 1,
                             tUobject_attr.m_bandwidth,
                             converted_act);
        if(SSP_OK != iRet)
        {
            PORT_LOG_ERR("sspLinePortSet failed. Port:%u Bandwidth:%u Action:%u",
                        usPort, tUobject_attr.m_bandwidth, converted_act);
        }
    }

    return iRet;
}
