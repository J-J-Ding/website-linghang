/*
描述：设置LX1891INA电口的速度和自协商
输入：usPortId - 物理端口号，ulSpeed - 速度，ulAutoNeg - 自协商状态
输出：成功返回SSP_OK，失败返回错误码
*/
INT32 sspPortSpeedAndNegLX1891INASet(UINT16 usPortId, UINT32 ulSpeed, UINT32 ulAutoNeg)
{
    INT32 ret = 0;
    PORT_LOG_NOTICE("phyPort %d is LX1891INA", usPortId);

    /* 联特电口不论配置1000M还是100M，都配置为SGMII模式，因此都传入1000M */
    if(SSP_OK != Supp_SetPortSerdesMode(usPortId,PORT_SPEED_1000))
    {
        PORT_LOG_ERR("HW copper phyPort:%d Speed %d set serdes mode fail!", usPortId, ulSpeed);
        return SSP_ERROR;
    }

    if (PORT_SPEED_1000 == ulSpeed)
    {
        if(SSP_OK != Supp_SetPortSpeed(usPortId,ulSpeed))
        {
            PORT_LOG_ERR("HW copper phyPort:%d Speed %d set fail!", usPortId, ulSpeed);
            return SSP_ERROR;
        }
    }

    if (PORT_AUTONEG_ENABLE == ulAutoNeg)
    {
        sspPortCopperAutoNegSet(usPortId, ulSpeed, ulAutoNeg);
    }
    else
    {
        if (PORT_SPEED_1000 == ulSpeed)
        {
            ret = sspPortCopper1000NoNegSet(usPortId, ulSpeed, ulAutoNeg);
            if (SSP_OK != ret)
            {
                PORT_LOG_ERR("HW copper phyPort:%d sspPortCopper1000NoNegSet set fail!", usPortId);
                return SSP_ERROR;
            }
            PORT_LOG_NOTICE("phyPort %d set 1000BASE_X", usPortId);
        }
        else
        {
            ret = sspPortCopper100NoNegSet(usPortId, ulSpeed, ulAutoNeg);
            ret |= Supp_SetPortSpeed(usPortId,PORT_SPEED_100);
            if (SSP_OK != ret)
            {
                PORT_LOG_ERR("HW copper phyPort:%d SPEED_100 set fail!", usPortId);
                return SSP_ERROR;
            }
            PORT_LOG_NOTICE("phyPort %d set SPEED_100", usPortId);
        }
    }

    return SSP_OK;
}

/*
描述：设置端口速度和自协商
输入：usPortId - 物理端口号，ulSpeed - 速度，ulAutoNeg - 自协商状态
输出：成功返回SSP_OK，失败返回错误码
*/
INT32 sspPortSpeedAndNegSet(UINT16 usPortId, UINT32 ulSpeed, UINT32 ulAutoNeg)
{
    INT32  ret = SSP_OK;
    UINT32 ulCfgNeg = 0;
    SSP_PORT_INFO_S *pstPortInfo = sspPortInfoGet(usPortId);

    if ((ulSpeed < PORT_SPEED_10) || (ulSpeed > PORT_SPEED_100000))
    {
        PORT_LOG_ERR("Phyport %d Speed %d is invalid!", usPortId, ulSpeed);
        return SSP_ERROR;
    }

    if ((ulAutoNeg < PORT_AUTONEG_ENABLE) || (ulAutoNeg > PORT_AUTONEG_DISABLE))
    {
        PORT_LOG_ERR("Phyport %d AutoNeg %d is invalid!", usPortId, ulAutoNeg);
        return SSP_ERROR;
    }

    if ((PORT_AUTONEG_ENABLE == ulAutoNeg) && (PORT_SPEED_1000 != ulSpeed))
    {
        ulCfgNeg = PORT_AUTONEG_DISABLE;
    }
    else
    {
        ulCfgNeg = ulAutoNeg;
    }

    pstPortInfo->ulSfpType = sspPortSfpTypeGet(pstPortInfo->usPortId);
    if (PORT_COPPER_SFP == pstPortInfo->ulSfpType)
    {
        ret = sspPortSpeedAndNegCopperSet(usPortId, ulSpeed, ulCfgNeg);
    }
    else
    {
        ret = sspPortSpeedAndNegFiberSet(usPortId, ulSpeed, ulCfgNeg);
    }

    if (SSP_OK != ret)
    {
        PORT_LOG_ERR("PhyPort %d Speed %d AutoNeg %d set fail!", usPortId, ulSpeed, ulAutoNeg);
        return ret;
    }

    pstPortInfo->ulAutoNeg = ulCfgNeg;
    pstPortInfo->ulSpeed = ulSpeed;
    return SSP_OK;
}

*/
INT32 SSP_PORT_InfoSet(UINT16 usPortId, SSP_PORT_INFO_S* pstPortInfo)
{
    INT32 ret = SSP_OK;

    if (NULL == pstPortInfo)
    {
        PORT_LOG_ERR("pstPortInfo is NULL!");
        return SSP_ERROR;
    }

    sspSemTake(g_ulPortSemId);

    /* 需要先配置端口类型，再配置其他属性 */
    ret |= sspPortTypeSet(usPortId, pstPortInfo->ulPortType);

    ret |= sspPortDuplexSet(usPortId, pstPortInfo->ulDuplex);

    ret |= sspPortSpeedAndNegSet(usPortId, pstPortInfo->ulSpeed, pstPortInfo->ulAutoNeg);

    ret |= sspPortFlowCtrlSet(usPortId, pstPortInfo->ulFlowCtrl);

    ret |= sspPortEnableSet(usPortId, pstPortInfo->ulEnable);

    sspSemGive(g_ulPortSemId);

    return ret;
}

    if ((usPortId < SSP_DEVICE_PHY_MIN) || (usPortId > SSP_DEVICE_PHY_MAX))
    {
        return SSP_ERROR;
    }

    ulSpeed = sspPortInfoGet(usPortId)->ulSpeed;

    if ( (usPortId <= SSP_DEVICE_LINE_MAX) && (usPortId >= SSP_DEVICE_LINE_MIN) )
    {
        sspSemTake(g_ulLinePortBandwidthSemId);
        sspLinePortBandwidthGet(usPortId, &ulLinePortSpeed);
        *pulSpeed = ulLinePortSpeed/1000; /*convert Kb/s to Mb/s*/
        sspSemGive(g_ulLinePortBandwidthSemId);

        if (0 == *pulSpeed)
        {
            *pulSpeed = SSP_PORT_SPEED_1000;/*默认1000M，避免分母为0*/
        }

        return SSP_OK;
    }

    if (SSP_DEVICE_TRUNK_MAX == usPortId) /*EOO背板连接端口*/
    {
        ulSpeed = PORT_SPEED_10000;
    }
    else if ((usPortId >= SSP_DEVICE_TRUNK_MIN) && (usPortId < SSP_DEVICE_TRUNK_MAX)) /*普通聚合端口*/
    {
        *pulSpeed = SSP_Trunk_SpeedGet(usPortId - SSP_DEVICE_TRUNK_MIN + 1);
        return SSP_OK;
    }

    *pulSpeed = sspGetSpeed(ulSpeed);

    return SSP_OK;
}

/*
描述：获取指定聚合组的总带宽
输入：usGroup - 聚合组ID
输出：返回聚合组总带宽（单位：Mbps）
 */
UINT32 SSP_Trunk_SpeedGet(UINT16 usGroup)
{
    UINT16 usPortId         = 0;
    UINT16 usActivePortId   = 0;
    UINT16 usActivePortNum  = 0;
    UINT32 ulBandwidth      = 0;
    TRUNK_GROUP_S *pTrunkList = NULL;

    pTrunkList = sspOuterGetTrunkHandle(usGroup);
    if (NULL == pTrunkList)
    {
        PORT_LOG_ERR("sspOuterGetTrunkHandle NULL");
        return SSP_PORT_SPEED_1000;
    }

    for (usPortId = SSP_DEVICE_CLIENT_MIN; usPortId <= SSP_DEVICE_CLIENT_MAX; usPortId++)
    {
        if (SSP_BIT_TEST(pTrunkList->m_tActiveMemberList, usPortId-1) == SSP_TRUE)
        {
            usActivePortId = usPortId;
            usActivePortNum++;
        }
    }

    if (usActivePortNum >0)
    {
        SSP_PORT_SpeedGet(usActivePortId, &ulBandwidth);
        return (ulBandwidth*usActivePortNum);
    }
    else
    {
        return SSP_PORT_SPEED_1000;
    }
}

/*
描述：设置端口类型和SFP类型
输入：pstPortInfo - 端口信息结构体指针
输出：无
*/
VOID sspReportPortType(SSP_PORT_INFO_S *pstPortInfo)
{
    if(NULL == pstPortInfo)
    {
        return;
    }

    if(OP_ELECTRICAL_CONNECTOR == sspPortSfpTypeGet(pstPortInfo->usPortId))
    {
        pstPortInfo->ulPortType = PORT_COPPER;
        pstPortInfo->ulSfpType  = PORT_COPPER_SFP;
    }
    else if(PORT_SPEED_100 == pstPortInfo->ulSpeed)
    {
        pstPortInfo->ulPortType = PORT_FIBER_FE;
    }
    else if(PORT_SPEED_1000 == pstPortInfo->ulSpeed)
    {
        pstPortInfo->ulPortType = PORT_FIBER_GE;
    }
    else if(PORT_SPEED_10000 == pstPortInfo->ulSpeed)
    {
        pstPortInfo->ulPortType = PORT_FIBER_XGE;
    }

    return;
}
