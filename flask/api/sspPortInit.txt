/*          1. 清除端口延迟UP标志                                              */
/*          2. 遍历所有客户端端口设置端口信息                                  */

/*防止端口实际link 状态跟网管不一致*/
void sspPortClientEnable()
{
    UINT16 usPortId = 0;
    SSP_PORT_INFO_S *pstPortInfo = NULL;
    STATUS ret = SSP_OK;

    //延时UP标记位清除，所有面板端口都mac enable
    g_ucPortUpDelay = SSP_FALSE;

    for(usPortId = SSP_DEVICE_CLIENT_MIN; usPortId <= SSP_DEVICE_CLIENT_MAX; usPortId++)
    {
        pstPortInfo = sspPortInfoGet(usPortId);

        ret = SSP_PORT_InfoSet(usPortId, pstPortInfo);
        if(SSP_OK != ret)
        {
            PORT_LOG_ERR(" port %d set portInfo error",usPortId);
            continue;
        }
        pstPortInfo = NULL;
    }

    g_ucDelyUPFlag = SSP_FALSE;
    PORT_LOG_NOTICE("=== ALL Client  enable ");
    return;
}

/*          2. 遍历所有客户端端口禁用MAC功能                                   */

void sspPortClientDisable()
{
    UINT16 usPortId = 0;

    //初始化时，先将所有端口都disable，为了D-LAG或者LAG，在本端上电过程中，让对端先不倒换
    for(usPortId = SSP_DEVICE_CLIENT_MIN; usPortId <= SSP_DEVICE_CLIENT_MAX; usPortId++)
    {
        SSP_HW_PORT_PortMacEnableSet(usPortId, PORT_ADMIN_DISABLE);
    }
    /*为了应对LAG和D-LAG在单板初始化的时候，端口震荡，设置延迟UP标记位*/
    g_ucPortUpDelay = SSP_TRUE;
    g_ucDelyUPFlag = SSP_TRUE;
    PORT_LOG_ERR(" === ALL Client mac disable  ");
    return;
}

VOID sspPortDefaulteGet(UINT16 usPortId, SSP_PORT_INFO_S *pstPortInfo)
{
    INT32 ret = SSP_OK;
    UINT16 usAttr = 0;
    PortAttribute *ptPortAttributes = NULL;

    SSP_MEMSET(pstPortInfo, sizeof(SSP_PORT_INFO_S), 0, sizeof(SSP_PORT_INFO_S));

    pstPortInfo->usPortId   = usPortId;
    pstPortInfo->ulEnable   = PORT_ADMIN_ENABLE;
    pstPortInfo->ulLoopback = PORT_LOOPBACK_NULL;
    pstPortInfo->ulPortType = PORT_FIBER_GE;
    pstPortInfo->ulAutoNeg  = PORT_AUTONEG_ENABLE;
    pstPortInfo->ulDuplex   = PORT_DUPLEX_FULL;
    pstPortInfo->ulSpeed    = PORT_SPEED_1000;
    pstPortInfo->ulFlowCtrl = PORT_FLOWCTRL_RX_TX_DISABLE;

    for (usAttr = 0; usAttr < MAX_ATTRIBUTES; usAttr++)
    {
        ptPortAttributes = devicePortAttr2VlaueGet(usAttr);

        switch (ptPortAttributes->id)
        {
            case CODE_PORT_TYPE:
                pstPortInfo->ulPortType = ptPortAttributes->value;
                break;
            case CODE_PORT_SPEED:
                pstPortInfo->ulSpeed = ptPortAttributes->value;
                break;
            case CODE_PORT_NEGOTIATION:
                pstPortInfo->ulAutoNeg = ptPortAttributes->value;
                break;
            case CODE_PORT_DUPLEX:
                pstPortInfo->ulDuplex = ptPortAttributes->value;
                break;
            case CODE_PORT_FLOW_CONTROL:
                pstPortInfo->ulFlowCtrl = ptPortAttributes->value;
                break;
            default:
                // Handle unknown attribute ID if necessary
                break;
        }
    }

    return;
}

INT32 sspPortInfoInit()
{
    INT32 ret = SSP_OK;
    UINT16 usPortId = 0;
    SSP_PORT_INFO_S stPortInfo = {0};
    SSP_PORT_INFO_S *ptPortInfo = NULL;

    for (usPortId = SSP_DEVICE_CLIENT_MIN; usPortId <= SSP_DEVICE_CLIENT_MAX; usPortId++)
    {
        sspPortDefaulteGet(usPortId, &stPortInfo);
        ret |= SSP_PORT_InfoSet(usPortId, &stPortInfo);

        ptPortInfo = sspPortInfoGet(usPortId);
        ptPortInfo->usPortId    = usPortId;
        ptPortInfo->ulLink      = PORT_LINK_DOWN;
        ptPortInfo->ulSfpType   = OP_UNKNOW_CONNECTOR;
        ptPortInfo->ulSfpStat   = PORT_SFP_OFFLINE;
    }

    return ret;
}

/*  返回值 : 初始化结果 (INT32)                                                */
/*  描  述 : 端口模块初始化入口函数                                            */
/*          1. 创建端口互斥信号量                                             */
/*          2. 初始化硬件端口模块                                              */
/*          3. 创建端口管理任务和统计任务                                      */
/*          4. 注册事件回调函数                                                */
/*  作  者 :                                                                   */
/*  版  本 :                                                                   */
/*  日  期 :                                                                   */

INT32 SSP_PORT_Init(VOID)
{
    INT32 ret = SSP_OK;

    /*端口模块互斥信号量初始化*/
    if (SSP_OK != sspPortSemIdCreate())
    {
        return SSP_ERROR;
    }
    /*端口模块带宽互斥信号量初始化*/
    if (SSP_OK != sspPortBandwidthSemIdCreate())
    {
        return SSP_ERROR;
    }

    /* 端口性能表项创建和初始化 */
    ret = sspPortStatCurCreate();
    if(SSP_OK != ret)
    {
        return SSP_ERROR;
    }
    ret = sspPortStatPreCreate();
    if(SSP_OK != ret)
    {
        return SSP_ERROR;
    }

    sspPortStatHistoryClearAll();

    sspPortRegisterCallBack();

    EventMgr_EventRegister(SSP_EVENT_PORT_UP, SSP_MODULE_PORT, (void *)sspPortLinkCallback);
    EventMgr_EventRegister(SSP_EVENT_PORT_DOWN, SSP_MODULE_PORT, (void *)sspPortLinkCallback);

    ret = SSP_HW_PORT_Init();
    if (SSP_OK != ret)
    {
        PORT_LOG_ERR("SSP_HW_PORT_Init Error!\n");
        return SSP_ERROR;
    }
    sspPortClientDisable();

    /* 软件表项初始化 */
    sspPortInfoClear();
    ret = sspPortInfoInit();
    if (SSP_OK != ret)
    {
        PORT_LOG_ERR("sspPortInfoInit Error!\n");
    }

    /* 配置表项初始化 */
    sspAllPortCfgInit();
    sspAllPortLoopbackCfgInit();

    ret |= sspTaskCreate((UINT8 *)SSP_TASK_PORT_SCAN_NAME,
                          sspAppPriGet((INT8 *)SSP_TASK_PORT_SCAN_NAME),
                          sspAppStackSizeGet((INT8 *)SSP_TASK_PORT_SCAN_NAME),
                          sspPortManageTask,
                          0,
                          &s_ulPortScanTaskId);
    if (SSP_OK != ret)
    {
        return SSP_ERROR;
    }

    /* 端口流量计数采集任务 */
    ret = sspTaskCreate((UINT8 *)SSP_TASK_PORT_STAT_NAME,
                         sspAppPriGet((INT8 *)SSP_TASK_PORT_STAT_NAME),
                         sspAppStackSizeGet((INT8 *)SSP_TASK_PORT_STAT_NAME),
                         sspPortStatTask,
                         0,
                         &s_ulPortStatTaskId);
    if (SSP_OK != ret)
    {
        return SSP_ERROR;
    }

    return SSP_OK;
}```
